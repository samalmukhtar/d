الأمر الوصف
sudo apt update && sudo apt upgrade -y تحديث النظام بالكامل
sudo apt install [package] تثبيت حزمة برمجية
sudo apt remove [package] إزالة حزمة مثبتة
ls -l عرض الملفات بتفاصيل كاملة
cp -r [source] [dest] نسخ مجلد كامل
mv [old] [new] إعادة تسمية ملف/مجلد
ping google.com فحص اتصال الشبكة
ifconfig عرض معلومات واجهات الشبكة
netstat -tuln عرض المنافذ المفتوحة
ps aux عرض جميع العمليات الجارية
kill -9 [PID] إنهاء عملية قسريًا
chmod 755 [file] تغيير صلاحيات الملف
chown [user]:[group] [file] تغيير ملكية الملف
nano [file] فتح ملف في محرر نانو
vim [file] فتح ملف في محرر فيم
useradd [username] إنشاء مستخدم جديد
passwd [username] تغيير كلمة مرور مستخدم
systemctl start [service] تشغيل خدمة النظام
systemctl stop [service] إيقاف خدمة النظام
df -h عرض مساحة التخزين المستخدمة
free -m عرض استخدام الذاكرة
uname -a عرض معلومات النظام الأساسية
ssh [user]@[host] الاتصال عن بعد بخادم
scp [file] [user]@[host]:[path] نسخ ملفات عبر SSH
wget [URL] تنزيل ملف من الإنترنت

grep 'text' [file] البحث عن نص داخل ملف
find / -name [filename] البحث عن ملف في النظام
history عرض سجل الأوامر
alias ll='ls -alF' إنشاء اسم مختصر لأمر
